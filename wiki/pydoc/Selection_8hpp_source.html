158 template<typename WTYPE>
159 void set_selection_parameters_from_slice(size_t i,const extract<slice> &slice,
160  const WTYPE &o,
161  std::vector<size_t> &offset,
162  std::vector<size_t> &stride,
163  std::vector<size_t> &shape)
164 {
165  //now we have to investigate the components of the
166  //slice
167  boost::python::ssize_t start = 0;
168  extract<size_t> __start(slice().start());
169  if(__start.check()) start = __start();
170 
171  boost::python::ssize_t step = 1;
172  extract<size_t> __step(slice().step());
173  if(__step.check()) step = __step();
174 
175  boost::python::ssize_t stop = o.shape()[i];
176  extract<boost::python::ssize_t> __stop(slice().stop());
177  if(__stop.check()) stop = __stop();
178 
179  //configure the selection
180  offset[i] = start;
181  stride[i] = step;
182 
183  boost::python::ssize_t res = (stop-start)%step;
184  shape[i] = (stop-start-res)/step;
185 }

201 template<typename WTYPE>
202 Selection create_selection(const WTYPE &o,const tuple &t)
203 {
204  //setup selection parameters
205  std::vector<size_t> offset(o.rank());
206  std::vector<size_t> stride(o.rank());
207  std::vector<size_t> shape(o.rank());
208 
209  //the number of elements in the tuple must not be equal to the
210  //rank of the field. This is due to the fact that the tuple can contain
211  //one ellipsis which spans over several dimensions.
212 
213  bool has_ellipsis = false;
214  size_t ellipsis_size = 0;
215  if(len(t) > (boost::python::ssize_t)o.rank())
216  throw_cdma_exception<ShapeNotMatchExceptionImpl>(
217  "Tuple with indices, slices, and ellipsis is longer than the "
218  "rank of the field - something went wrong here",
219  "template<typename WTYPE> Selection create_selection(const "
220  "WTYPE &o,const tuple &pyselection)");
221  else if(len(t) != (boost::python::ssize_t)o.rank())
222  {
223  //here we have to fix the size of an ellipsis
224  ellipsis_size = o.rank()-(len(t)-1);
225  }
226 
227  /*this loop has tow possibilities:
228  -> there is no ellipse and the rank of the field is larger than the size of
229  the tuple passed. In this case an IndexError will occur. In this case we
230  know immediately that a shape error occured.
231  */
232  for(size_t i=0,j=0;i<o.rank();i++,j++)
233  {
234  //manage a single index
235  extract<size_t> index(t[j]);
236  if(index.check())
237  {
238  set_selection_parameters_from_index(i,index,offset,stride,shape);
239  continue;
240  }
241 
242  //manage a slice
243  extract<slice> s(t[j]);
244  if(s.check())
245  {
246  set_selection_parameters_from_slice(i,s,o,offset,stride,shape);
247  continue;
248  }
249 
250  //if we came until here the only possible type the tuples component can
251  //be an ellipsis. If it is not an ellipsis an exception will be thrown.
252  const object &ellipsis = t[j];
253  if(Py_Ellipsis != ellipsis.ptr())
254  throw_PyTypeError("A selection must contain only indices,slices"
255  "and a single ellipsis!");
256 
257  //assume here that the object is an ellipsis - this is a bit difficult
258  //to handle as we do not know over how many
259  if(!has_ellipsis)
260  {
261  has_ellipsis = true;
262  while(i<j+ellipsis_size)
263  {
264  stride[i] = 1;
265  offset[i] = 0;
266  shape[i] = o.shape()[i];
267  i++;
268  }
269  }else{
270  //throw index error exception here since there is only one ellipsis
271  //allowed in the tuple
272  }
273  }
274 
275  //once we are done with looping over all elemnts in the tuple we need
276  //to adjust the selection to take into account an ellipsis
277  if((ellipsis_size) && (!has_ellipsis))
278  throw_cdma_exception<ShapeNotMatchExceptionImpl>(
279  "Selection rank does not match DataItem rank",
280  "template<typename WTYPE> Selection create_selection(const "
281  "WTYPE &o,const tuple &pyselection)");
282 
283  return Selection(offset,stride,shape);
284 }
