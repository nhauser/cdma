00040     private:

00082         {
00083             if(this != &i){
00084                 _container = i._container;

00086             }
00087             return *this;
00088         }

00090         //---------------------------------------------------------------------
00097         value_type next()
00098         {
00099             //check if iteration is still possible
00100             if(_state == this->_container->size())
00101             {
00102                 throw_PyStopIteration("iteration finished");
00103                 return(value_type());
00104             }
00105 
00106             //return the current object 
00107             value_type item((*(this->_container))[this->_state]);
00108             //increment the iterator
00109             this->_state++;
00110 
00111             return item;
00112         }
00113 
00114         //----------------------------------------------------------------------
00116         object __iter__()
00117         {
00118             return object(this);
00119         }
