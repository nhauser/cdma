00148 template<typename WTYPE> object __getitem__(WTYPE &o,object &selection)
00149 {
00150     //if the data object itself is scalar we can only return a scalar value
00151     //in this case we ignore all arguments to __getitem__
00152     if((o.shape().size()==1)||(o.type() == TypeID::STRING))
00153             return read_scalar_data(o);
00154 
00155     //ok - we have a multidimensional data object. Now it depends on the 
00156     //selection object of what will be returned. The selection object can either
00157     //be a list or tuple or a single python object from which the selection must
00158     //be assembled. 
00159 
00160     Selection sel;
00161     if(PyTuple_Check(selection.ptr()))
00162         sel = create_selection(o,tuple(selection));
00163     else
00164         sel = create_selection(o,make_tuple<object>(selection));
00165 
00166     //std::cout<<sel<<std::endl;
00167 
00168     //now we have the selection we need to read data - as CDMA actually does not
00169     //support strides others than 1 we have to fix this here
00170     std::vector<size_t> offset(sel.offset());
00171     std::vector<size_t> shape(sel.shape());
00172     for(size_t i=0;i<sel.rank();i++)
00173         shape[i] *= sel.stride()[i];
00174 
00175     //read data
00176     ArrayWrapper array = o.get(offset,shape);
00177 
00178     //now we need to convert the array to a numpy array
00179     return cdma2numpy_array(array,true);
00180 }

00193 template<typename WTYPE> object read_scalar_data(WTYPE &o)
00194 {
00195     switch(o.type())
00196     {
00197         case TypeID::BYTE: return object(o.template get<int8_t>());
00198         case TypeID::UBYTE: return object(o.template get<uint8_t>());
00199         case TypeID::SHORT: return object(o.template get<int16_t>());
00200         case TypeID::USHORT: return object(o.template get<uint16_t>());
00201         case TypeID::INT: return object(o.template get<int32_t>());
00202         case TypeID::UINT: return object(o.template get<uint32_t>());
00203         case TypeID::LONG: return object(o.template get<int64_t>());
00204         case TypeID::ULONG: return object(o.template get<uint64_t>());
00205         case TypeID::FLOAT: return object(o.template get<float>());
00206         case TypeID::DOUBLE: return object(o.template get<double>());
00207         case TypeID::STRING: return object(o.template get<std::string>());
00208         default:
00209             throw_PyTypeError("Data type not supported!");
00210 
00211     };
00212 
00213     return object(); //return value only to avoid compiler warnings 
00214                      //this code will never be reached.
00215 }
