137 template<typename WTYPE> object __getitem__(WTYPE &o,object &selection)
138 {
139  //if the data object itself is scalar we can only return a scalar value
140  //in this case we ignore all arguments to __getitem__
141  if((o.shape().size()==1)||(o.type() == STRING))
142  return read_scalar_data(o);
143 
144  //ok - we have a multidimensional data object. Now it depends on the
145  //selection object of what will be returned. The selection object can either
146  //be a list or tuple or a single python object from which the selection must
147  //be assembled.
148 
149  Selection sel;
150  if(PyTuple_Check(selection.ptr()))
151  sel = create_selection(o,tuple(selection));
152  else
153  sel = create_selection(o,make_tuple<object>(selection));
154 
155  //std::cout<<sel<<std::endl;
156 
157  //now we have the selection we need to read data - as CDMA actually does not
158  //support strides others than 1 we have to fix this here
159  std::vector<size_t> offset(sel.offset());
160  std::vector<size_t> shape(sel.shape());
161  for(size_t i=0;i<sel.rank();i++)
162  shape[i] *= sel.stride()[i];
163 
164  //read data
165  ArrayWrapper array = o.get(offset,shape);
166 
167  //now we need to convert the array to a numpy array
168  return cdma2numpy_array(array,true);
169 }

182 template<typename WTYPE> object read_scalar_data(WTYPE &o)
183 {
184  switch(o.type())
185  {
186  case BYTE: return object(o.template get<int8_t>());
187  case UBYTE: return object(o.template get<uint8_t>());
188  case SHORT: return object(o.template get<int16_t>());
189  case USHORT: return object(o.template get<uint16_t>());
190  case INT: return object(o.template get<int32_t>());
191  case UINT: return object(o.template get<uint32_t>());
192  case LONG: return object(o.template get<int64_t>());
193  case ULONG: return object(o.template get<uint64_t>());
194  case FLOAT: return object(o.template get<float>());
195  case DOUBLE: return object(o.template get<double>());
196  case STRING: return object(o.template get<std::string>());
197  default:
198  throw_PyTypeError("Data type not supported!");
199 
200  };
201 
202  return object(); //return value only to avoid compiler warnings
203  //this code will never be reached.
204 }
