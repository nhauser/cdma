00134 template<typename WTYPE> object __getitem__(WTYPE &o,object &selection)
00135 {
00136     //if the data object itself is scalar we can only return a scalar value
00137     //in this case we ignore all arguments to __getitem__
00138     if((o.shape().size()==1)||(o.type() == TypeID::STRING))
00139             return read_scalar_data(o);
00140 
00141     //ok - we have a multidimensional data object. Now it depends on the 
00142     //selection object of what will be returned. The selection object can either
00143     //be a list or tuple or a single python object from which the selection must
00144     //be assembled. 
00145 
00146     Selection sel;
00147     if(PyTuple_Check(selection.ptr()))
00148         sel = create_selection(o,tuple(selection));
00149     else
00150         sel = create_selection(o,make_tuple<object>(selection));
00151 
00152     //std::cout<<sel<<std::endl;
00153 
00154     //now we have the selection we need to read data - as CDMA actually does not
00155     //support strides others than 1 we have to fix this here
00156     std::vector<size_t> offset(sel.offset());
00157     std::vector<size_t> shape(sel.shape());
00158     for(size_t i=0;i<sel.rank();i++)
00159         shape[i] *= sel.stride()[i];
00160 
00161     //read data
00162     ArrayWrapper array = o.get(offset,shape);
00163 
00164     //now we need to convert the array to a numpy array
00165     return cdma2numpy_array(array,true);
00166 }

00178 template<typename WTYPE> object read_scalar_data(WTYPE &o)
00179 {
00180     switch(o.type())
00181     {
00182         case TypeID::BYTE: return object(o.template get<int8_t>());
00183         case TypeID::UBYTE: return object(o.template get<uint8_t>());
00184         case TypeID::SHORT: return object(o.template get<int16_t>());
00185         case TypeID::USHORT: return object(o.template get<uint16_t>());
00186         case TypeID::INT: return object(o.template get<int32_t>());
00187         case TypeID::UINT: return object(o.template get<uint32_t>());
00188         case TypeID::LONG: return object(o.template get<int64_t>());
00189         case TypeID::ULONG: return object(o.template get<uint64_t>());
00190         case TypeID::FLOAT: return object(o.template get<float>());
00191         case TypeID::DOUBLE: return object(o.template get<double>());
00192         case TypeID::STRING: return object(o.template get<std::string>());
00193         default:
00194             throw_PyTypeError("Data type not supported!");
00195 
00196     };
00197 
00198     return object(); //return value only to avoid compiler warnings 
00199                      //this code will never be reached.
00200 }
